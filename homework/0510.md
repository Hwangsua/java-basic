package com.day02.quiz;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.Scanner;

import javax.swing.JOptionPane;

public class Quiz02 {
	
	private final String JDBC_DRIVER = "org.mariadb.jdbc.Driver"; // Driver 클래스 풀네임 (JQFN)
	private final String DB_URL = "jdbc:mysql://localhost:3306/testDB?"
								+ "useUnicode=true"
								+ "&characterEncoding=utf8";
	
	private final String USERNAME = "root";
	private final String PASSWORD = "1234"; 
	
	private String name, grade;
	private int no, kr, en, ma;
	private float avg;
	
	
	public StringBuilder getMsg(ResultSet rs) {
		StringBuilder msg = new StringBuilder();
		try {
			if(!rs.next()) { // 커서 이동하여 비어있는지 확인
				return msg.append("해당하는 학생이 없습니다.");
			}
			rs.beforeFirst(); // 커서 원위치
			while(rs.next()) { // 커서 이동하여 비어있는지 확인
				msg.append("이름 : ").append(rs.getString(2))
				.append("\n학번 : ").append(rs.getInt(1))
				.append("\n국어\\영어\\수학 : ").append(rs.getInt(3))
				.append("\\").append(rs.getInt(4))
				.append("\\").append(rs.getInt(5))
				.append("\n평균 : ").append(rs.getFloat(6))
				.append("\n성적 : ").append(rs.getString(7))
				.append("\n등록일자 : ").append(rs.getString(8)).append("\n\n");
			}
		} catch (SQLException e) {
			
			e.printStackTrace();
		}
		return msg;
	}
	
	public void runSql(String sql, int flag) {
		Connection conn = null;
		PreparedStatement ps = null;
		ResultSet rs = null;
		
		try {
			
			Class.forName(JDBC_DRIVER);
			conn = DriverManager.getConnection(DB_URL,USERNAME,PASSWORD);
			ps = conn.prepareStatement(sql);
			
			if(flag == 1) { // 등록
				
				ps.setString(1, name); 
				ps.setInt(2, kr);
				ps.setInt(3, en);
				ps.setInt(4, ma);
				ps.setFloat(5, avg);
				ps.setString(6, grade);
				ps.execute();
				
			}else if(flag == 2) { // 전체 학생 확인
				rs = ps.executeQuery(); // ResultSet 객체 결과값 담는다.
				JOptionPane.showMessageDialog(null, getMsg(rs));
				
			}else if(flag == 3) {
				ps.setString(1, "%"+name+"%");
				rs = ps.executeQuery();
				JOptionPane.showMessageDialog(null, getMsg(rs));
				
			}else if(flag == 4) {
				ps.setInt(1, no);
				rs = ps.executeQuery();
				JOptionPane.showMessageDialog(null, getMsg(rs));
				
			}else if(flag == 5) {
				rs = ps.executeQuery();
				JOptionPane.showMessageDialog(null, getMsg(rs));
			}
			
			
			
		} catch (Exception e) {
			e.printStackTrace();
		}finally {
			try {
				if(rs!=null) {rs.close();}
				if(ps!=null) {ps.close();}
				if(conn!=null) {conn.close();}
			} catch(Exception e){
				e.printStackTrace();
			}
		}
	}
	
	// 테이블 생성
	public void createTable() {
	
		String sql  = "CREATE TABLE if NOT EXISTS Student("
				+ "no INT PRIMARY KEY AUTO_INCREMENT,"
				+ "name VARCHAR(40) NOT NULL,"
				+ "kr INT(3) CHECK (0<= kr AND kr<=100),"
				+ "en INT(3) CHECK (0<= kr AND kr<=100),"
				+ "ma INT(3) CHECK (0<= kr AND kr<=100),"
				+ "avg FLOAT(5,2),"
				+ "grade VARCHAR(1) DEFAULT 'F',"
				+ "regdate DATETIME DEFAULT CURRENT_TIMESTAMP)";
		runSql(sql,0);
		System.out.println("테이블생성");
	}
	
	// 학생 등록
	public void insertStudent(String name, int kr, int en, int ma) {
		String sql = "INSERT INTO student VALUES(null,?,?,?,?,?,?,DEFAULT)";
		this.name = name;
		this.kr = kr;
		this.en = en;
		this.ma = ma;
		this.avg = (float) ((kr+en+ma)/3);
		setGrade(avg);
		runSql(sql, 1);
	}
	
	// 등급 설정
	public void setGrade(float avg) {
		int key = (int)(avg/10);
		String tmp;
		
		switch (key) {
		case 10: case 9: { tmp = "A"; break; }
		case 8: { tmp = "B"; break; }
		case 7: { tmp = "C"; break; }
		case 6: { tmp = "D"; break; }
		default: tmp = "F";}
		this.grade = tmp;
	}
	
	// 모든 학생 보기
	public void selectStudents() {
		String sql = "SELECT * FROM student";
		runSql(sql,2);
	}
	
	// 이름으로 학생 검색
	public void selectName(String name) {
		String sql = "SELECT * FROM student WHERE name LIKE ?";
		// setString 하면 ''가 자동으로 생성되기 때문에 ?만 적는다.
		this.name = name;
		runSql(sql, 3);	
	}
	
	// 학번으로 학생 검색 
	public void selectNumber(int no) {
		String sql = "SELECT * FROM student WHERE no = ?";
		this.no = no;
		runSql(sql, 4);
	}
	
	public void selectHighScore() {
		String sql = "SELECT * FROM student WHERE avg = (SELECT MAX(avg) FROM student)";
		runSql(sql,5);
	}
	public Quiz02() {
		createTable();
	}

	public static void main(String[] args) {
		

		Quiz02 stTable = new Quiz02();
		String menu = "< MENU >\n"
				+"1. 학생 등록\n"
				+"2. 모든 학생 보기\n"
				+"3. 이름으로 검색\n"
				+"4. 학번으로 검색\n"
				+"5. 1등 보기\n"
				+"6. 종료\n";
		
		loop :
		while(true) {
			String select = JOptionPane.showInputDialog(menu);
			
			switch (select) {
			case "1": {
				stTable.insertStudent(JOptionPane.showInputDialog("이름 입력하세요."), 
									Integer.parseInt(JOptionPane.showInputDialog("국어점수")), 
									Integer.parseInt(JOptionPane.showInputDialog("영어점수")),
									Integer.parseInt(JOptionPane.showInputDialog("수학점수")));
				break;
			}
			case "2": {
				stTable.selectStudents();
				break;	
			}
			case "3": {
				stTable.selectName(JOptionPane.showInputDialog("이름을 입력하세요."));
				break;	
			}
			case "4": {
				stTable.selectNumber(Integer.parseInt(JOptionPane.showInputDialog("학번을 입력하세요.")));
				break;	
			}
			case "5": {
				stTable.selectHighScore();
				break;	
			}
			case "6": {
				break loop;	
			}
			default:
				JOptionPane.showMessageDialog(null, "다시 입력하세요.");
			}
			
		}		
		
	}

}
