# 과제 (04.12)

## 1. setters / getters

```java
package day16.quiz;

class Student {
	private String name;
	private int kr, en, ma;
	private double avg;
	private boolean pass;
	private String grade;
	
	// 1. 생성자 (여러분 마음대로 여러 개 만들기) 
	public Student(){
		this(null,0,0,0);
	}
	public Student(String name){
		this(name,0,0,0);
	}
	public Student(String name, int kr, int en, int ma){
		this.setName(name);
		this.setKr(kr);
		this.setEn(en);
		this.setMa(ma);
	}
	// 2. getters (형식에 맞게)
	
	public String getName() {
		return name;
	}
	
	public int getKr() {
		return kr;
	}
	public int getEn() {
		return en;
	}
	public int getMa() {
		return ma;
	}

	public double getAvg() {
		return avg;
	}
	public boolean isPass() {
		return pass;
	}
	public String getGrade() {
		return grade;
	}
	
	// 3. setters (재료를 넣으면 검열을 거쳐 올바른 데이터만 들어감)
	//  0) name : 이름
	//  1) kr, en, ma : 0 점 이상 100점 이하만 저장 가능, 그 외 0점 
	//  2) avg : (인자값 받지 않고, kr, en, ma만 가지고 계산되도록)
	//  3) pass : 60점 이상이면 true
	//  4) grade : ABCDF 중 1개로
	
	
	public void setName(String name) {
		this.name = name;
	}
	// 점수를 수정하면 평균이 바뀜 => 평균 다시 구한다.
	public void setKr(int kr) {
		this.kr = (0<=kr && kr<=100)? kr:0;
		setAvg();
	}
	public void setEn(int en) {
		this.en = (0<=en && en<=100)? en:0;
		setAvg();
	}
	public void setMa(int ma) {
		this.ma = (0<=ma && ma<=100)? ma:0;
		setAvg();
	}
	// 점수 바뀌면 setAvg()하므로 공개될 필요 없다.
	// 외부에서 실행할 필요없는 것들을 다 private
	private void setAvg() {
		avg = (kr+en+ma)/3.0;
		setPass();
		setGrade();
	}
	private void setPass() {
		pass = avg>=60;
		// pass = (avg>=60)?true:false;
	}
	private void setGrade() {
		switch((int)avg/10) {
		case 10:case 9:
			grade = "A";
			break;
		case 8:
			grade = "B";
			break;
		case 7:
			grade = "C";
			break;
		case 6:
			grade = "D";
			break;
		default:
			grade = "F";	
		}
	}
	public void printData() {
		System.out.println("이름: "+name);
		System.out.println("국/영/수: "+kr+"/"+en+"/"+ma);
		System.out.println("평균: "+avg);
		System.out.println("학점: "+grade);
		System.out.println("결과: "+(pass?"합격":"불합격"));
	}
	
}
public class Quiz01 {
	public static void main(String[] args) {
		// Student 테스트 (마음대로)
		Student[] st = {
						 new Student(),
						 new Student("뽀로로"),
						 new Student("크롱",80,60,200)
						};
		
		for(Student s : st) {
			s.printData();
			System.out.println();
		}
		
		st[2].setMa(60);
		System.out.println("-- 수정 후 --");
		for(Student s : st) {
			s.printData();
			System.out.println();
		}
		
		
		
	}
}
```
